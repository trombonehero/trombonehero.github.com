{
if (args == NULL) error(OK);
// Count 'em: don't bother with shared memory for zero arguments.
size_t argc = 0;
while ((argc <= MAX_ARGC) && (args[argc] != NULL)) argc++;
if (argc == 0) error(OK);
if (argc > MAX_ARGC) error(E2BIG);
// Calculate total memory required to store arguments + terminators.
size_t lengths[argc];
size_t total_length = 1; // null terminator
for (size_t i = 0; i < argc; i++) {
char *arg = args[i];
if (arg == NULL) error(EINVAL);
size_t len = strnlen(arg, MAX_ARGLEN);
if (len == MAX_ARGLEN) error(EFBIG);
lengths[i] = len;
total_length += (len + 1); // null terminator
}
// Create a shared memory segment, map it and copy the arguments over.
int fd = shm_open(SHM_ANON, O_RDWR, 0);
if (fd < 0) return -1;
if (ftruncate(fd, total_length) < 0) return -1;
if (fcntl(fd, F_SETFD, 0) == -1) return -1; // don't close on exec
char *mem = mmap(NULL, total_length, PROT_READ | PROT_WRITE, 0, fd, 0);
if (mem == MAP_FAILED) return -1;
char *head = mem;
for (size_t i = 0; i < argc; i++)
head = stpncpy(head, args[i], lengths[i] + 1) + 1;
munmap(mem, total_length);
return fd;
}
